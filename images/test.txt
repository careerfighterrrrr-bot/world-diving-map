            /* 幻の島の画像を表示。名前を images/background.jpg にしてアップロードしてください */
            background: url('images/background.jpg') no-repeat center center;
            background-size: cover;

            display: flex; justify-content: center; align-items: center;
            transition: opacity 1.5s ease-out;

    <div id="splash-screen">

    <div id="map"></div>

    <script src="https://unpkg.com/leaflet@1.9.4/dist/leaflet.js"></script>
    <script>
        // 1. 地図の初期設定
        const map = L.map('map').setView([24.40, 124.15], 11);
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png').addTo(map);

        // 2. ポイントデータ（ウミガメの画像もここに追加）
        const divePoints = [
            {
                name: "幻の島（浜島）",
                coords: [24.405, 124.030],
                desc: "干潮時にしか現れない砂の島。初心者OK！",
                img: "images/background.jpg"
            },
            {
                name: "ウミガメ・ポイント",
                coords: [24.455, 124.110],
                desc: "遭遇率高め。水深5〜10mで安全に楽しめます。",
                img: "images/turtle.jpg" // ウミガメの画像
            }
        ];

        // 3. ピンを刺す
        divePoints.forEach(p => {
            const content = `
                <div style="width:200px">
                    <b>${p.name}</b><br>${p.desc}<br>
                    <img src="${p.img}" class="popup-img">
                </div>`;
            L.marker(p.coords).addTo(map).bindPopup(content);
        });

        // 4. 3秒後に起動画面を消す
        setTimeout(() => {
            const splash = document.getElementById('splash-screen');
            splash.style.opacity = '0';
            setTimeout(() => splash.style.display = 'none', 1500);
        }, 3000);
    </script>
